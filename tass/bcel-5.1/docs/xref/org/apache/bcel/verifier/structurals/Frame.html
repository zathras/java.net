61  /***
62   * This class represents a JVM execution frame; that means,
63   * a local variable array and an operand stack.
64   *
65   * @version $Id: Frame.html,v 1.1.1.1 2006/05/19 19:25:35 billf Exp $
66   * @author <A HREF="http://www.inf.fu-berlin.de/~ehaase"/>Enver Haase</A>

69  public class Frame{
70  
71  	/***
72  	 * For instance initialization methods, it is important to remember
73  	 * which instance it is that is not initialized yet. It will be
74  	 * initialized invoking another constructor later.
75  	 * NULL means the instance already *is* initialized.
76  	 */
77  	protected static UninitializedObjectType _this;
78  
79  	/***
80  	 *
81  	 */
82  	private LocalVariables locals;
83  
84  	/***
85  	 *
86  	 */
87  	private OperandStack stack;
88  
89  	/***
90  	 *
91  	 */
92  	public Frame(int maxLocals, int maxStack){
93  		locals = new LocalVariables(maxLocals);
94  		stack = new OperandStack(maxStack);
95  	}
96  
97  	/***
98  	 *
99  	 */
100 	public Frame(LocalVariables locals, OperandStack stack){
101 		this.locals = locals;
102 		this.stack = stack;
103 	}
104 
105 	/***
106 	 *
107 	 */
108 	protected Object clone(){
109 		Frame f = new Frame(locals.getClone(), stack.getClone());
110 		return f;
111 	}
112 
113 	/***
114 	 *
115 	 */
116 	public Frame getClone(){
117 		return (Frame) clone();
118 	}
119 
120 	/***
121 	 *
122 	 */
123 	public LocalVariables getLocals(){
124 		return locals;
125 	}
126 
127 	/***
128 	 *
129 	 */
130 	public OperandStack getStack(){
131 		return stack;
132 	}
133 
134 	/***
135 	 *
136 	 */
137 	public boolean equals(Object o){
138 		if (!(o instanceof Frame)) return false; // implies "null" is non-equal.
139 		Frame f = (Frame) o;
140 		return this.stack.equals(f.stack) && this.locals.equals(f.locals);
141 	}
142 
143 	/***
144 	 * Returns a String representation of the Frame instance.
145 	 */
146 	public String toString(){
147 		String s="Local Variables:\n";
148 		s += locals;
149 		s += "OperandStack:\n";
150 		s += stack;
151 		return s;
152 	}
