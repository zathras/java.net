57  import  org.apache.bcel.Constants;
58  import  java.io.*;
59  
60  /***
61   * This class is derived from <em>Attribute</em> and represents a constant 
62   * value, i.e., a default value for initializing a class field.
63   * This class is instantiated by the <em>Attribute.readAttribute()</em> method.
64   *
65   * @version $Id: ConstantValue.html,v 1.1.1.1 2006/05/19 19:21:52 billf Exp $
66   * @author  <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
67   * @see     Attribute
68   */
69  public final class ConstantValue extends Attribute {
70    private int constantvalue_index;
71  
72    /***
73     * Initialize from another object. Note that both objects use the same
74     * references (shallow copy). Use clone() for a physical copy.
75     */
76    public ConstantValue(ConstantValue c) {
77      this(c.getNameIndex(), c.getLength(), c.getConstantValueIndex(),
78  	 c.getConstantPool());
79    }
80  
81    /***
82     * Construct object from file stream.
83     * @param name_index Name index in constant pool
84     * @param length Content length in bytes
85     * @param file Input stream
86     * @param constant_pool Array of constants
87     * @throw IOException
88     */
89    ConstantValue(int name_index, int length, DataInputStream file,
90  		ConstantPool constant_pool) throws IOException
91    {
92      this(name_index, length, (int)file.readUnsignedShort(), constant_pool);
93    }    
94  
95    /***
96     * @param name_index Name index in constant pool
97     * @param length Content length in bytes
98     * @param constantvalue_index Index in constant pool
99     * @param constant_pool Array of constants
100    */  
101   public ConstantValue(int name_index, int length, 
102 		       int constantvalue_index,
103 		       ConstantPool constant_pool)
104   {
105     super(Constants.ATTR_CONSTANT_VALUE, name_index, length, constant_pool);
106     this.constantvalue_index = constantvalue_index;
107   }    
108 
109   /***
110    * Called by objects that are traversing the nodes of the tree implicitely
111    * defined by the contents of a Java class. I.e., the hierarchy of methods,
112    * fields, attributes, etc. spawns a tree of objects.
113    *
114    * @param v Visitor object
115    */
116   public void accept(Visitor v) {
117     v.visitConstantValue(this);
118   }    
119   /***
120    * Dump constant value attribute to file stream on binary format.
121    *
122    * @param file Output file stream
123    * @throws IOException
124    */ 
125   public final void dump(DataOutputStream file) throws IOException
126   {
127     super.dump(file);
128     file.writeShort(constantvalue_index);
129   }    
130   /***
131    * @return Index in constant pool of constant value.
132    */  
133   public final int getConstantValueIndex() { return constantvalue_index; }    
134 
135   /***
136    * @param constantvalue_index.
137    */
138   public final void setConstantValueIndex(int constantvalue_index) {
139     this.constantvalue_index = constantvalue_index;
140   }    
141 
142   /***
143    * @return String representation of constant value.
144    */ 
145   public final String toString() {
146     Constant c = constant_pool.getConstant(constantvalue_index);
147 	
148     String   buf;
149     int    i;
150 
151     // Print constant to string depending on its type
152     switch(c.getTag()) {
153     case Constants.CONSTANT_Long:    buf = "" + ((ConstantLong)c).getBytes();    break;
154     case Constants.CONSTANT_Float:   buf = "" + ((ConstantFloat)c).getBytes();   break;
155     case Constants.CONSTANT_Double:  buf = "" + ((ConstantDouble)c).getBytes();  break;
156     case Constants.CONSTANT_Integer: buf = "" + ((ConstantInteger)c).getBytes(); break;
157     case Constants.CONSTANT_String:  
158       i   = ((ConstantString)c).getStringIndex();
159       c   = constant_pool.getConstant(i, Constants.CONSTANT_Utf8);
160       buf = "\"" + Utility.convertString(((ConstantUtf8)c).getBytes()) + "\"";
161       break;
162 
163     default:
164       throw new IllegalStateException("Type of ConstValue invalid: " + c);
165     }
166 
167     return buf;
168   }
169 
170   /***
171    * @return deep copy of this attribute
172    */
173   public Attribute copy(ConstantPool constant_pool) {
174     ConstantValue c = (ConstantValue)clone();
175     c.constant_pool = constant_pool;
176     return c;
177   }
